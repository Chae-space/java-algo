### 문제를 해결한 코드
```java



//나의 코드

import java.util.Scanner;

public class Main {

    public String Solution(int num, int[] arr) {
        String answer = "";
        arr[0] = 1;
        arr[1] = 1;
        for (int i = 2; i < num; i++) {
            arr[i] = arr[i - 2] + arr[i - 1];
        }
        for (int i : arr)
            answer += i + " ";
        return answer;
       }

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int num = sc.nextInt();
        int[] arr = new int[num];
        Main M = new Main();
        System.out.println(M.Solution(num, arr));

    }
}

```
