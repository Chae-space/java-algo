### 구현 배경 지식
- 숫자 뒤집기 알고리즘 :
  숫자를 뒤집으려면 숫자를 10으로 나눈 나머지를 계속 더해준다.
  더할때 기존 숫자에 곱하기 10을 해줘야 자릿수를 하나씩 넘길 수 있다

### 문제를 해결한 코드
```java
import java.util.*;
class Main {	
	public boolean isPrime(int num){       //소수인지 아닌지 확인
		if(num==1) return false;           //1은 소수가 아니므로 false
		for(int i=2; i<num; i++){          
			if(num%i==0) return false;     //이 조건식이 참이면 i는 num의 약수이므로 num은 소수가 아님
		}
		return true;
	}

	public ArrayList<Integer> solution(int n, int[] arr){
		ArrayList<Integer> answer = new ArrayList<>();
		for(int i=0; i<n; i++){
			int tmp=arr[i];
			int res=0;
			while(tmp>0){                 //숫자 뒤집기
				int t=tmp%10;
				res=res*10+t;
				tmp=tmp/10;
			}
			if(isPrime(res)) answer.add(res);
		}
		return answer;
	}
	public static void main(String[] args){
		Main T = new Main();
		Scanner kb = new Scanner(System.in);
		int n=kb.nextInt();
		int[] arr=new int[n];
		for(int i=0; i<n; i++){
			arr[i]=kb.nextInt();
		}
		for(int x : T.solution(n, arr)){
			System.out.print(x+" ");
		}
	}
}




```
